  <!-- PDF.js library from the Mozilla hosted demo (works fine for pages) -->

  <script src="https://mozilla.github.io/pdf.js/build/pdf.js"></script>

  <script>

    // File name must match exactly

    const url = 'Varun-Gautam.pdf';


    // PDF.js config

    pdfjsLib.GlobalWorkerOptions.workerSrc = 'https://mozilla.github.io/pdf.js/build/pdf.worker.js';


    let pdfDoc = null, pageNum = 1, scale = 1.25;

    const canvas = document.getElementById('pdfCanvas');

    const ctx = canvas.getContext('2d');


    function renderPage(num) {

      pdfDoc.getPage(num).then(function(page) {

        const viewport = page.getViewport({ scale: scale });

        canvas.width = viewport.width;

        canvas.height = viewport.height;


        const renderContext = {

          canvasContext: ctx,

          viewport: viewport

        };

        page.render(renderContext).promise.then(function () {

          document.getElementById('pageIndicator').textContent = `Page ${pageNum} / ${pdfDoc.numPages}`;

        });

      });

    }


    // Load the PDF

    pdfjsLib.getDocument(url).promise.then(function(pdf) {

      pdfDoc = pdf;

      // render first page

      renderPage(pageNum);

    }).catch(function (err) {

      console.error('Error loading PDF:', err);

      document.getElementById('canvasContainer').innerHTML =

        '<div style="padding:28px;font-size:1rem;color:#222">Unable to render PDF in browser. <a href="Varun-Gautam.pdf" target="_blank" rel="noopener">Open PDF directly</a></div>';

    });


    // Controls

    document.getElementById('prevBtn').addEventListener('click', function(){

      if (pageNum <= 1) return;

      pageNum--;

      renderPage(pageNum);

    });

    document.getElementById('nextBtn').addEventListener('click', function(){

      if (!pdfDoc || pageNum >= pdfDoc.numPages) return;

      pageNum++;

      renderPage(pageNum);

    });

    document.getElementById('zoomIn').addEventListener('click', function(){

      scale = Math.min(scale + 0.25, 3);

      renderPage(pageNum);

    });

    document.getElementById('zoomOut').addEventListener('click', function(){

      scale = Math.max(scale - 0.25, 0.5);

      renderPage(pageNum);

    });


    // Show Open link direct to file

    document.getElementById('openPdf').href = url;

  </script>
